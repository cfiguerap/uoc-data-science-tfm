def plot_by_size(real, prediction, t):
    l = len(real)
    plt.figure(figsize=(18, 12))
    plt.title('Prediccin')
    plt.xlabel('Tiempo')
    plt.ylabel('Consumo')
    plt.plot(real[l-t:l], color='blue')
    plt.plot(prediction[l-t:l], color='red')
    plt.legend(['real', 'prediction'])
    plt.show()
